# cal

```bash
cal        # календарь
ncal       # перевернутый календарь
ncal -w    # с номером недели
cal 2018   # показать весь год
cal -m12   # показать месяц
cal -3     # прошлый, текущий и следующий месяц
cal -h     # не выделять текущую дату
```
